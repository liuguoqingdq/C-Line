#### 创建表
```SQL
CREATE TABLE 【表名】 (
  -- ===== 1) 字段定义 =====
  【字段名1】 【数据类型】 【是否为空】 【默认值】 【自增】 COMMENT '【字段注释】',
  【字段名2】 【数据类型】 【是否为空】 【默认值】            COMMENT '【字段注释】',
  ...
  【字段名N】 【数据类型】 【是否为空】 【默认值】            COMMENT '【字段注释】',

  -- ===== 2) 主键 =====
  PRIMARY KEY (【主键字段1】[, 【主键字段2】 ...]),

  -- ===== 3) 唯一约束（可选）=====
  UNIQUE KEY 【唯一索引名】 (【唯一字段1】[, 【唯一字段2】 ...]),

  -- ===== 4) 普通索引（可选）=====
  KEY 【普通索引名1】 (【索引字段1】[, 【索引字段2】 ...]),
  KEY 【普通索引名2】 (【索引字段1】[, 【索引字段2】 ...]),

  -- ===== 5) 外键（可选）=====
  CONSTRAINT 【外键名】
    FOREIGN KEY (【本表外键字段1】[, 【本表外键字段2】 ...])
    REFERENCES 【被引用表名】 (【被引用字段1】[, 【被引用字段2】 ...])
    ON DELETE 【删除规则RESTRICT/CASCADE/SET NULL/NO ACTION】
    ON UPDATE 【更新规则RESTRICT/CASCADE/SET NULL/NO ACTION】

) ENGINE=【存储引擎InnoDB等】
  DEFAULT CHARSET=【字符集utf8mb4等】
  COLLATE=【排序规则可选】
  COMMENT='【表注释】';
```

**修改表**
```SQL
//添加一个字段
ALTER TABLE 表名
ADD salary DOUBLE(10,2)//默认追加到最后
/*
在 MySQL 里写 `DOUBLE(M, D)` 这两个参数的含义是：

- **M**：总位数（**总数字个数**，包含小数点左边 + 右边的数字，但**不**包含小数点和负号）
    
- **D**：小数位数（小数点**后**的位数，scale 
*/
ALTER TABLE 表名
ADD phone_number varchar(20) FIRST;//放在第一位

ALTER TABLE 表名
ADD email varchar(20) AFTER emp_name;//在某个字段后

//修改一个字段
ALTER TABLE 表名
MODIFY emp_name varchar(25);

//默认值
ALTER TABLE 表名
MODIFY emp_name varchar(25) DEFAULT 'aaa';

//重命名
ALTER TABLE 表名
CHANGE salary month_salary DOUBLE(10,2);

//删除一个字段
ALTER TABLE 表名
DROP COLUMN email；

//重命名表
RENAME TABLE 旧表名
TO 新表名


//删除表
DROP TABLE IF EXISTS 表名；
 
 
//清空表
TRUNCATE TABLE 表名；
```
